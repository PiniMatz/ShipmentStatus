async function waitForFingerprint(timeout = 10000) {
  const startTime = Date.now();
  
  while (true) {
	  console.log("while");
    const fingerprint = $.Storage.get("fingerprint");
    
    if (fingerprint && fingerprint !== null && fingerprint !== "null") {
      return fingerprint;
    }
    
    
    if (Date.now() - startTime > timeout) {
      console.warn("Таймаут ожидания fingerprint");
      return null;
    }
    await new Promise(resolve => setTimeout(resolve, 100));
  }
}

		function dateDiff(date1, date2) {
           date1 = date1.split(".");
		   date1 = date1[1]+'/'+date1[0]+'/'+date1[2];
		   
		   date2 = date2.split(".");
		   date2 = date2[1]+'/'+date2[0]+'/'+date2[2];
		   
		   date2 = date2.replace(".", "/");
		   date2 = date2.replace(".", "/");
		 
		   var dtFrom = date1; 
		   var dtTo = date2;
		 
		   var dt1 = new Date(dtFrom);
		   var dt2 = new Date(dtTo);
		   var diff = dt2.getTime() - dt1.getTime();
		   var days = diff/(1000 * 60 * 60 * 24); 
		   return  Math.round(days);
		}
		function numberof(numberof, value, suffix) {
			var numberof = Math.abs(numberof);
			var keys = [2, 0, 1, 1, 1, 2];
			var mod = numberof % 100;
			var suffix_key = mod > 4 && mod < 20 ? 2 : keys[Math.min(mod%10, 5)];
			return value+suffix[suffix_key];
		 }

	function findBootstrapEnvironment() {
		var envs = ['xs', 'sm', 'md', 'lg', 'xl'];

		var el = document.createElement('div');
		document.body.appendChild(el);

		var curEnv = envs.shift();

		for (var env of envs.reverse()) {
			el.classList.add(`d-${env}-none`);

			if (window.getComputedStyle(el).display === 'none') {
				curEnv = env;
				break;
			}
		}

		document.body.removeChild(el);
    return curEnv;
}
